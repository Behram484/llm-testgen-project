package net.sourceforge.schemaspy.util;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

public class InflectionTest_v1 {
 @Test
 void testPluralize() {
 assertEquals("people", Inflection.pluralize("person"));
 assertEquals("men", Inflection.pluralize("man"));
 assertEquals("children", Inflection.pluralize("child"));
 assertEquals("sexes", Inflection.pluralize("sex"));
 assertEquals("moves", Inflection.pluralize("move"));
 }

 @Test
 void testSingularize() {
 assertEquals("person", Inflection.singularize("people"));
 assertEquals("man", Inflection.singularize("men"));
 assertEquals("child", Inflection.singularize("children"));
 assertEquals("sex", Inflection.singularize("sexes"));
 assertEquals("move", Inflection.singularize("moves"));
 }

 @Test
 void testUncountable() {
 assert(Inflection.isUncountable("equipment"));
 assert(Inflection.isUncountable("information"));
 assert(Inflection.isUncountable("rice"));
 assert(Inflection.isUncountable("money"));
 assert(Inflection.isUncountable("species"));
 assert(Inflection.isUncountable("series"));
 assert(Inflection.isUncountable("fish"));
 assert(Inflection.isUncountable("sheep"));
 }
}
