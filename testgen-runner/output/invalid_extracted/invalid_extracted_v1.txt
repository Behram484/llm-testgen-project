package utils;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

public class UtilsTest_v1 {
 
 @Test
 void testEncodePath() {
 String path = "/home/user";
 assertEquals("--/-home-user", Utils.encodePath(path));
 
 path = "C:\\Windows";
 assertEquals("C:-Windows", Utils.encodePath(path));
 }
 
 @Test
 void testEncodePathAsIdentifier() {
 String path = "/home/user";
 assertEquals("_2fhome2fuser", Utils.encodePathAsIdentifier(path));
 
 path = "C:\\Windows";
 assertEquals("C__3aWindows", Utils.encodePathAsIdentifier(path));
 }
 
 @Test
 void testAbsoluteURL() {
 String url = "http://www.example.com";
 assert (Utils.absoluteURL(url));
 
 url = "/home/user";
 assert (!Utils.absoluteURL(url));
 }
 
 @Test
 void testPseudoAbsoluteURL() {
 String url = "http://www.example.com";
 assert (Utils.pseudoAbsoluteURL(url));
 
 url = "/home/user";
 assert (Utils.pseudoAbsoluteURL(url));
 }
 
 @Test
 void testNChars() {
 int n = 5;
 char c = 'a';
 assertEquals("aaaaa", Utils.nChars(n, c));
 
 n = 3;
 c = 'b';
 assertEquals("bbb", Utils.nChars(n, c));
 }
}
